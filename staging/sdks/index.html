üìñ Quick Start Example (Python)

from seatrace import SeaTraceClient

# Initialize client
client = SeaTraceClient(api_key='YOUR_API_KEY')

# Get all vessels
vessels = client.seaside.get_vessels()

# Create catch record
catch = client.deckside.create_catch({
    'vesselId': 'vessel-001',
    'species': 'Pacific Salmon',
    'weight': 1250.5,
    'location': {'lat': 34.0522, 'lon': -118.2437}
})

print(f"Catch ID: {catch['id']}")
print(f"QR Code: {catch['qrCode']}")
